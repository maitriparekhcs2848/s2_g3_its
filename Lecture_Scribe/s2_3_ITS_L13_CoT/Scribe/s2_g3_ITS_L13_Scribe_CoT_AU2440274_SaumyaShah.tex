\section*{Q1. Division of Dishes into Platters}

\subsection*{Problem Setting}

There are 20 distinct dishes divided into 4 groups of 5 dishes each.
\begin{itemize}
    \item Each platter contains exactly 5 dishes.
    \item Order within a platter does not matter.
    \item Order of platters does not matter.
\end{itemize}

\subsection*{(a) Total Number of Ways}

Total permutations of 20 dishes:
\[
20!
\]

Adjust for overcounting:
\[
\text{Total ways} = \frac{20!}{(5!)^4 \cdot 4!}
\]

\subsection*{(b) Probability Each Platter Has One Cuisine}

Favorable outcomes = 1

\[
P = \frac{1}{\frac{20!}{(5!)^4 4!}}
= \frac{(5!)^4 4!}{20!}
\]

\section*{Q3. Conditional Probability}

Given:
\[
P(L)=0.15,\quad P(E)=0.25,\quad P(L\cap E)=0.08
\]

\[
P(E^c)=0.75
\]

\[
P(L\cup E)=0.15+0.25-0.08=0.32
\]

\[
P(L^c \cap E^c)=1-0.32=0.68
\]

\[
P(L^c|E^c)=\frac{0.68}{0.75}=0.907
\]

\[

\]

\[

\[

\section*{Q8. Jury Decision (Binomial Model)}

If defendant is guilty:

\[
\sum_{i=8}^{12} \binom{12}{i}\theta^i(1-\theta)^{12-i}
\]

If innocent:

\[
\sum_{i=5}^{12} \binom{12}{i}\theta^i(1-\theta)^{12-i}
\]

Overall probability:

\[
\alpha\sum_{i=8}^{12}\binom{12}{i}\theta^i(1-\theta)^{12-i}
+(1-\alpha)\sum_{i=5}^{12}\binom{12}{i}\theta^i(1-\theta)^{12-i}
\]

\section*{Q10. Reliability of Systems}

5-component system:
\[
10p^3(1-p)^2+5p^4(1-p)+p^5
\]

3-component system:
\[
3p^2(1-p)+p^3
\]

Condition:
\[
3(p-1)^2(2p-1)>0
\]

\[
p>\frac{1}{2}
\]

General case difference:

\[
\binom{2k-1}{k}p^k(1-p)^k(2p-1)
\]

\[
>0 \iff p>\frac{1}{2}
\]
