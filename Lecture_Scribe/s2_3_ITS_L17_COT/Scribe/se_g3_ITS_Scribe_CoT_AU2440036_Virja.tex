\section{Why Study Joint Random Variables?}

In many real-world situations we study \textbf{two or more random variables together} instead of a single variable. This helps us analyze the \textbf{relationship between variables}.

\subsection{Examples}

\begin{itemize}
\item Height and weight of giraffes
\item IQ and birthweight of children
\item Exercise frequency and heart disease rate
\item Air pollution level and respiratory illness rate
\item Number of Facebook friends and age of members
\end{itemize}

These examples show that multiple variables may influence each other.

\subsection{Smart Transportation}

Two random variables:

\[
X = \text{Vehicle Speed}
\]

\[
Y = \text{Traffic Density}
\]

Question: Can speed be high when traffic density is also high?

Applications:

\begin{itemize}
\item Adaptive traffic signals
\item Congestion prediction
\item Autonomous vehicle routing
\end{itemize}

\subsection{Wireless Network Performance}

\[
X = \text{Signal-to-Noise Ratio (SNR)}
\]

\[
Y = \text{Data Throughput}
\]

Even if SNR is high, throughput may not always be high due to congestion or interference.

Applications:

\begin{itemize}
\item 5G scheduling
\item Adaptive modulation
\item Network planning
\end{itemize}

\subsection{Other Examples}

\begin{itemize}
\item Weather prediction: $X=$ Rainfall, $Y=$ Temperature
\item Drone delivery: $X=$ Wind Speed, $Y=$ Delivery Time
\item Rolling two dice: $X$ and $Y$ are values of the dice
\end{itemize}

Key idea: Random variables together have a \textbf{joint distribution}.

\section{Discrete Case -- Joint PMF}

Suppose $X$ and $Y$ are discrete random variables.

\[
X = \{x_1, x_2, ..., x_n\}
\]

\[
Y = \{y_1, y_2, ..., y_m\}
\]

The pair $(X,Y)$ takes values in

\[
\{(x_1,y_1),(x_1,y_2),...,(x_n,y_m)\}
\]

\subsection{Definition}

The joint PMF is

\[
p(x_i,y_j) = P(X = x_i, Y = y_j)
\]

This represents the probability that both events occur together.

\section{Joint PMF Table}

\begin{center}
\begin{tabular}{c|ccc}
X/Y & $y_1$ & $y_2$ & $y_m$ \\ \hline
$x_1$ & $p(x_1,y_1)$ & $p(x_1,y_2)$ & $p(x_1,y_m)$ \\
$x_2$ & $p(x_2,y_1)$ & $p(x_2,y_2)$ & $p(x_2,y_m)$ \\
$x_n$ & $p(x_n,y_1)$ & $p(x_n,y_2)$ & $p(x_n,y_m)$ \\
\end{tabular}
\end{center}

\subsection{Properties}

\[
0 \le p(x_i,y_j) \le 1
\]

\[
\sum_{i=1}^{n} \sum_{j=1}^{m} p(x_i,y_j) = 1
\]

\section{Example -- Rolling Two Dice}

Roll two fair dice.

\[
X = \text{value on first die}
\]

\[
T = \text{sum of both dice}
\]

There are 36 equally likely outcomes.

\[
P(i,j) = \frac{1}{36}
\]

Observation:

\begin{itemize}
\item Table entries are either $0$ or $\frac{1}{36}$
\item $X$ and $T$ are not independent
\end{itemize}

\section{Continuous Case -- Joint PDF}

\begin{center}
\begin{tabular}{c|c}
Discrete & Continuous \\
\hline
Joint PMF & Joint PDF \\
Sums & Integrals \\
\end{tabular}
\end{center}

If

\[
X \in [a,b]
\]

\[
Y \in [c,d]
\]

then

\[
(X,Y) \in [a,b] \times [c,d]
\]

\subsection{Definition}

A function $f(x,y)$ is a joint PDF if

\[
P((X,Y) \text{ in small region}) \approx f(x,y)\,dx\,dy
\]

\subsection{Properties}

\[
f(x,y) \ge 0
\]

\[
\int_c^d \int_a^b f(x,y)\,dx\,dy = 1
\]

\section{Example}

Given

\[
f(x,y) = 4xy
\]

\[
0 \le x \le 1, \quad 0 \le y \le 1
\]

Check normalization:

\[
\int_0^1 \int_0^1 4xy\,dx\,dy
\]

First integrate w.r.t $x$

\[
\int_0^1 4xy\,dx = 4y \int_0^1 x\,dx
\]

\[
=4y\left[\frac{x^2}{2}\right]_0^1
\]

\[
=2y
\]

Now integrate w.r.t $y$

\[
\int_0^1 2y\,dy
\]

\[
=[y^2]_0^1 = 1
\]

Thus it is a valid joint PDF.

\subsection{Finding Probability}

Event

\[
A = \{X<0.5, Y>0.5\}
\]

\[
P(A) = \int_0^{0.5} \int_{0.5}^{1} 4xy\,dy\,dx
\]

First integrate w.r.t $y$

\[
\int_{0.5}^{1}4xy\,dy
\]

\[
=4x\left[\frac{y^2}{2}\right]_{0.5}^{1}
\]

\[
=2x(1-0.25)
\]

\[
=1.5x
\]

Now integrate w.r.t $x$

\[
\int_0^{0.5} 1.5x\,dx
\]

\[
=1.5\left[\frac{x^2}{2}\right]_0^{0.5}
\]

\[
=\frac{3}{16}
\]

\section{Joint CDF}

The joint cumulative distribution function is

\[
F(x,y) = P(X \le x, Y \le y)
\]

\section{Continuous Case}

If $f(x,y)$ is the joint PDF

\[
F(x,y) = \int_c^y \int_a^x f(u,v)\,du\,dv
\]

Recover the PDF

\[
f(x,y) = \frac{\partial^2}{\partial x \partial y}F(x,y)
\]

\section{Discrete Case}

\[
F(x,y) = \sum_{x_i \le x} \sum_{y_j \le y} p(x_i,y_j)
\]

\section{Properties of Joint CDF}

\begin{itemize}
\item $F(x,y)$ is non-decreasing
\item $F(x,y) \rightarrow 0$ as $(x,y) \rightarrow (-\infty,-\infty)$
\item $F(x,y) \rightarrow 1$ as $(x,y) \rightarrow (\infty,\infty)$
\end{itemize}

\section{Quick Revision Formulas}

Joint PMF

\[
p(x_i,y_j) = P(X=x_i,Y=y_j)
\]

PMF property

\[
\sum_i \sum_j p(x_i,y_j) = 1
\]

Joint PDF

\[
P((X,Y)\in region) = \iint f(x,y)\,dx\,dy
\]

Normalization

\[
\int\int f(x,y)\,dx\,dy = 1
\]

Joint CDF

\[
F(x,y) = P(X \le x, Y \le y)
\]
