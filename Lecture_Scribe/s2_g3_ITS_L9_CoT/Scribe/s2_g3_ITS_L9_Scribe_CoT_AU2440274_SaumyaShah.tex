\section{Uniform Random Variable}

\subsection{Definition}

Let $X$ be a continuous random variable uniformly distributed over the interval $[a,b]$.

\subsubsection*{Probability Density Function (PDF)}

\[
f_X(x) =
\begin{cases}
\dfrac{1}{b-a}, & a \le x < b \\
0, & \text{otherwise}
\end{cases}
\]

\subsubsection*{Cumulative Distribution Function (CDF)}

\[
F_X(x) =
\begin{cases}
0, & x < a \\
\dfrac{x-a}{b-a}, & a \le x < b \\
1, & x \ge b
\end{cases}
\]

\subsection{Interpretation from Graphs}

\begin{itemize}
    \item The PDF is constant on $[a,b]$ with height $\dfrac{1}{b-a}$.
    \item The CDF increases linearly from $0$ at $x=a$ to $1$ at $x=b$.
\end{itemize}

\section{Example 1: Uniform Random Variable}

The phase $\Theta$ is uniformly distributed over $[0,2\pi)$.

\[
f_\Theta(\theta) =
\begin{cases}
\dfrac{1}{2\pi}, & 0 \le \theta < 2\pi \\
0, & \text{otherwise}
\end{cases}
\]

\subsection*{(a) Compute $\Pr(\Theta > 3\pi/4)$}

\[
\Pr(\Theta > 3\pi/4)
= \frac{2\pi - 3\pi/4}{2\pi}
= \frac{5\pi/4}{2\pi}
= \frac{5}{8}
\]

\subsection*{(b) Compute $\Pr(\Theta < \pi \mid \Theta > 3\pi/4)$}

Let:
\[
A = \{\Theta < \pi\}, \quad
B = \{\Theta > 3\pi/4\}
\]

\[
A \cap B = \left(3\pi/4, \pi\right)
\]

\[
\Pr(A \cap B)
= \frac{\pi - 3\pi/4}{2\pi}
= \frac{\pi/4}{2\pi}
= \frac{1}{8}
\]

\[
\Pr(B) = \frac{5}{8}
\]

\[
\Pr(A \mid B)
= \frac{\Pr(A \cap B)}{\Pr(B)}
= \frac{1/8}{5/8}
= \frac{1}{5}
\]

\subsection*{(c) Compute $\Pr(\cos\Theta < 1/2)$}

\[
\cos\Theta = \frac{1}{2}
\quad \Rightarrow \quad
\Theta = \frac{\pi}{3}, \; \frac{5\pi}{3}
\]

\[
\Pr\left(\cos\Theta < \frac{1}{2}\right)
= \frac{5\pi/3 - \pi/3}{2\pi}
= \frac{4\pi/3}{2\pi}
= \frac{2}{3}
\]

%------------------------------------------------
\section{Applications of Uniform Random Variable}

\begin{itemize}
    \item Phase of a sinusoidal signal when all phase angles between $0$ and $2\pi$ are equally likely.
    \item Random number generated by a computer between $0$ and $1$ for simulations.
    \item Arrival time of a user within a known time window assuming no time preference.
\end{itemize}

\section{Exponential Random Variable}

\subsection{Definition}

For $b>0$, the exponential random variable has:

\subsubsection*{Probability Density Function (PDF)}

\[
f_X(x) = \frac{1}{b} e^{-x/b} u(x)
\]

\subsubsection*{Cumulative Distribution Function (CDF)}

\[
F_X(x) = \left[1 - e^{-x/b}\right] u(x)
\]

\begin{itemize}
    \item The PDF decreases exponentially for $x \ge 0$.
    \item The CDF increases asymptotically toward $1$.
    \item Support: $x \ge 0$.
\end{itemize}

\section{Example 2: Exponential Random Variable}

Let:
\[
f_X(x) = e^{-x} u(x)
\]

Thus $b=1$.

\subsection*{(a) Compute $\Pr(3X < 5)$}

\[
3X < 5 \quad \Rightarrow \quad X < \frac{5}{3}
\]

\[
\Pr(3X < 5)
= \Pr\left(X < \frac{5}{3}\right)
= 1 - e^{-5/3}
\]

\subsection*{(b) Compute $\Pr(3X < y)$}

\[
3X < y \quad \Rightarrow \quad X < \frac{y}{3}
\]

\[
\Pr(3X < y)
= \Pr\left(X < \frac{y}{3}\right)
= 1 - e^{-y/3}
\]
